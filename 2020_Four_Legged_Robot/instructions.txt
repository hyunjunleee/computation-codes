void trot(int steps)
{
  int k = (steps - 2) / 2;
  for (int theta = 60; theta >= 30; theta--)
  {
    int t = 60 - theta;
    actuate(1, s / 2 + s * cos(2 * theta), -H - sqrt(3)*s / 2 + s * sin(2 * theta), 0);
    actuate(2, s / 2 + s * cos(2 * theta), -H - sqrt(3)*s / 2 + s * sin(2 * theta), 0);
    actuate(0, -s * t / 30, -H, 0);
    actuate(3, -s * t / 30, -H, 0);
    delay(5);
  }
  for (int i = 1; i <= k; i++)
  {
    for (int theta = 120; theta >= 60; theta--)
    {
      int t = 120 - theta;
      actuate(1, s - s * t / 30, -H, 0);
      actuate(2, s - s * t / 30, -H, 0);
      actuate(0, 2 * s * cos(theta), 2 * s * sin(theta) - H - sqrt(3)*s, 0);
      actuate(3, 2 * s * cos(theta), 2 * s * sin(theta) - H - sqrt(3)*s, 0);
      delay(5);
    }
    for (int theta = 120; theta >= 60; theta--)
    {
      int t = 120 - theta;
      actuate(0, s - s * t / 30, -H, 0);
      actuate(3, s - s * t / 30, -H, 0);
      actuate(1, 2 * s * cos(theta), 2 * s * sin(theta) - H - sqrt(3)*s, 0);
      actuate(2, 2 * s * cos(theta), 2 * s * sin(theta) - H - sqrt(3)*s, 0);
      delay(5);
    }
  }
  for (int theta = 60; theta >= 30; theta--)
  {
    int t = 60 - theta;
    actuate(1, s - s * t / 30, -H, 0);
    actuate(2, s - s * t / 30, -H, 0);
    actuate(0, -s / 2 + s * cos(2 * theta), -H - sqrt(3)*s / 2 + s * sin(2 * theta), 0);
    actuate(3, -s / 2 + s * cos(2 * theta), -H - sqrt(3)*s / 2 + s * sin(2 * theta), 0);
    delay(5);
  }
}

void translate(float steps)
{
  /*READ THISreplace z with x in 'trot'*/
  int k = (steps - 2) / 2;
  for (int theta = 60; theta >= 30; theta--)
  {
    int t = 60 - theta;
    actuate(1, 0, -H - sqrt(3)*s / 2 + s * sin(2 * theta), s / 2 + s * cos(2 * theta));
    actuate(2, 0, -H - sqrt(3)*s / 2 + s * sin(2 * theta), s / 2 + s * cos(2 * theta));
    actuate(0, 0, -H, -s * t / 30);
    actuate(3, 0, -H, -s * t / 30);
    delay(5);
  }
  for (int i = 1; i <= k; i++)
  {
    for (int theta = 120; theta >= 60; theta--)
    {
      int t = 120 - theta;

      actuate(0, 0, 2 * s * sin(theta) - H - sqrt(3)*s, 2 * s * cos(theta));
      actuate(3, 0, 2 * s * sin(theta) - H - sqrt(3)*s, 2 * s * cos(theta));
      delay(5);
    }
    for (int theta = 120; theta >= 60; theta--)
    {
      int t = 120 - theta;
      actuate(0, 0 , -H, s - s * t / 30);
      actuate(3, 0, -H, s - s * t / 30);

/*roll pitch yaw*/
void roll(float t)
{
  if (t > 0)
  {
    for (int theta = 0; theta <= t; theta++)
    {
      actuate(0, 0, -H * cos(theta) + W * sin(theta) / 2, W / 2 - H * sin(theta) - W * cos(theta) / 2);
      actuate(2, 0, -H * cos(theta) + W * sin(theta) / 2, W / 2 - H * sin(theta) - W * cos(theta) / 2);
      actuate(1, 0, -H * cos(theta) - W * sin(theta) / 2, -W / 2 - H * sin(theta) + W * cos(theta) / 2);
      actuate(3, 0, -H * cos(theta) - W * sin(theta) / 2, -W / 2 - H * sin(theta) + W * cos(theta) / 2);
      delay(15);
    }
  }
  if (t < 0)
  {
    for (int theta = 0; theta >= t; theta--)
    {
      actuate(0, 0, -H * cos(theta) + W * sin(theta) / 2, W / 2 - H * sin(theta) - W * cos(theta) / 2);
      actuate(2, 0, -H * cos(theta) + W * sin(theta) / 2, W / 2 - H * sin(theta) - W * cos(theta) / 2);
      actuate(1, 0, -H * cos(theta) - W * sin(theta) / 2, -W / 2 - H * sin(theta) + W * cos(theta) / 2);
      actuate(3, 0, -H * cos(theta) - W * sin(theta) / 2, -W / 2 - H * sin(theta) + W * cos(theta) / 2);
      delay(15);
    }
  }
}


void pitch(float t)
{
  if (t > 0)
  {
    for (int theta = 0; theta <= t; theta++)
    {
      actuate(0, -L / 2 + L * cos(theta) / 2 - H * sin(theta), -L * sin(theta) / 2 - H * cos(theta), 0);
      actuate(1, -L / 2 + L * cos(theta) / 2 - H * sin(theta), -L * sin(theta) / 2 - H * cos(theta), 0);
      actuate(2, L / 2 - L * cos(theta) / 2 - H * sin(theta), L * sin(theta) / 2 - H * cos(theta), 0);
      actuate(3, L / 2 - L * cos(theta) / 2 - H * sin(theta), L * sin(theta) / 2 - H * cos(theta), 0);
      delay(15);
    }
  }
  if (t < 0)
  {
    for (int theta = 0; theta >= t; theta--)
    {
      actuate(0, -L / 2 + L * cos(theta) / 2 - H * sin(theta), -L * sin(theta) / 2 - H * cos(theta), 0);
      actuate(1, -L / 2 + L * cos(theta) / 2 - H * sin(theta), -L * sin(theta) / 2 - H * cos(theta), 0);
      actuate(2, L / 2 - L * cos(theta) / 2 - H * sin(theta), L * sin(theta) / 2 - H * cos(theta), 0);
      actuate(3, L / 2 - L * cos(theta) / 2 - H * sin(theta), L * sin(theta) / 2 - H * cos(theta), 0);
      delay(15);
    }
  }
}


void yaw(float t)
{
  if (t > 0)
  {
    for (theta = 0; theta <= t; theta++)
    {
      actuate(0, -L / 2 + L * cos(theta) / 2 + W * sin(theta) / 2, -H, W / 2 + L * sin(theta) / 2 - W * cos(theta) / 2);
      actuate(1, -L / 2 + L * cos(theta) / 2 - W * sin(theta) / 2, -H, -W / 2 + L * sin(theta) / 2 + W * cos(theta) / 2);
      actuate(2, L / 2 - L * cos(theta) / 2 + W * sin(theta) / 2, -H, W / 2 - L * sin(theta) / 2 - W * cos(theta) / 2);
      actuate(3, L / 2 - L * cos(theta) / 2 - W * sin(theta) / 2, -H, -W / 2 - L * sin(theta) / 2 + W * cos(theta) / 2);
      delay(15);
    }
  }
  if (t < 0)
  {
    for (theta = 0; theta >= t; theta--)
    {
      actuate(0, -L / 2 + L * cos(theta) / 2 + W * sin(theta) / 2, -H, W / 2 + L * sin(theta) / 2 - W * cos(theta) / 2);
      actuate(1, -L / 2 + L * cos(theta) / 2 - W * sin(theta) / 2, -H, -W / 2 + L * sin(theta) / 2 + W * cos(theta) / 2);
      actuate(2, L / 2 - L * cos(theta) / 2 + W * sin(theta) / 2, -H, W / 2 - L * sin(theta) / 2 - W * cos(theta) / 2);
      actuate(3, L / 2 - L * cos(theta) / 2 - W * sin(theta) / 2, -H, -W / 2 - L * sin(theta) / 2 + W * cos(theta) / 2);
      delay(15);
    }
  }
}

/*inverse kinematics*/
void actuate(int a, float x, float y, float z)
{
  if (a == 1 || a == 3)
  {
    float x1 = x;
    float y1 = -sqrt(y * y + z * z - Lc * Lc);
    servo[a][0].write(acos(Lc * y1 * (y / Lc + z / y1) / (y1 * y1 + Lc * Lc)));
    servo[a][1].write(acos((Lf * Lf + x1 * x1 + y1 * y1 - Lt * Lt) / (2 * Lf * sqrt(x1 * x1 + y1 * y1))) - atan(x1 / y1));
    servo[a][2].write(asin((x ^ 2 + y ^ 2 + z ^ 2 - Lc ^ 2 - Lt ^ 2 - Lf ^ 2) / (2Lf * Lt)));
  }
  if (a == 2 || a == 4)
  {
    float x1 = x;
    float y1 = -sqrt(y * y + z * z - Lc * Lc);
    servo[a][0].write(acos(Lc * y1 * (y / Lc - z / y1) / (y1 * y1 + Lc * Lc)));
    servo[a][1].write(acos((Lf * Lf + x1 * x1 + y1 * y1 - Lt * Lt) / (2 * Lf * sqrt(x1 * x1 + y1 * y1))) - atan(x1 / y1));
    servo[a][2].write(asin((x ^ 2 + y ^ 2 + z ^ 2 - Lc ^ 2 - Lt ^ 2 - Lf ^ 2) / (2Lf * Lt)));
  }
}
